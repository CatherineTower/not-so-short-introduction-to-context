% This snippet defines a footer with page numbers. A name is assigned
% to it, and then, when section commands are configured, this name is
% assigned to the first page of part and chapter.

\definetext[ChapterFirstPage]
           [footer]
           [pagenumber]


%    conversion=I,

\setuphead
  [part]
  [placehead=yes,
   page=right,
   bodypartlabel=part,
   alternative=middle,
   style={\switchtobodyfont[36pt]\bf},
   color=purple,
   before={\strut\godown[4cm]},
   header=high,
   footer=ChapterFirstPage,]

\setuphead
  [title, chapter]
  [
    style={\switchtobodyfont[30pt]\bf},
    align=flushright,
    before={\blank[5*big, force]},
    after={\blank[2*big]},
    color=maincolor,
    sectionsegments=1:2,
    commandbefore={\blank[small]},
    header=high,
    footer=ChapterFirstPage,
  ]

\setuphead
  [section, subject]
  [
    page=yes,
    continue=no,      % nouvelle page même si derrière \startpart
    style={\bfc},
    before={\blank[2*big]},
    after=\blank,
    color=maincolor,
    sectionsegments=3:3,
  ]

\setuphead
  [subsection, subsubject]
  [
    style={\bfb},
    before=\blank,
    after=\blank,
    color=maincolor,
    sectionsegments=3:4,
  ]

\define[1]\PlacePoint{#1.}

\setuphead
  [subsubsection,subsubsubject]
  [
    sectionsegments=3:5,
%    conversion=A,
    style=\bfa,
    color=black,
%    numbercommand=,
  ]

\setuphead
  [subsubsection][numbercommand=\PlacePoint]
